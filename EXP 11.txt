#include <stdio.h>

int main() {
    int frames[10], pages[30];
    int n, m, pageFaults = 0;
    int i, j, k = 0, flag;

    printf("Enter number of frames: ");
    scanf("%d", &n);

    printf("Enter number of pages: ");
    scanf("%d", &m);

    printf("Enter page reference string:\n");
    for (i = 0; i < m; i++) {
        scanf("%d", &pages[i]);
    }

    // Initialize all frames to -1 (empty)
    for (i = 0; i < n; i++)
        frames[i] = -1;

    printf("\nPage\tFrames\t\tPage Fault\n");
    printf("----------------------------------\n");

    for (i = 0; i < m; i++) {
        flag = 0;

        // Check if page is already in a frame
        for (j = 0; j < n; j++) {
            if (frames[j] == pages[i]) {
                flag = 1; // page hit
                break;
            }
        }

        // If page not found -> page fault
        if (flag == 0) {
            frames[k] = pages[i]; // replace oldest page
            k = (k + 1) % n;      // move to next frame (FIFO)
            pageFaults++;
        }

        // Print frames
        printf("%d\t", pages[i]);
        for (j = 0; j < n; j++) {
            if (frames[j] == -1)
                printf("- ");
            else
                printf("%d ", frames[j]);
        }

        if (flag == 0)
            printf("\tPage Fault\n");
        else
            printf("\tNo Fault\n");
    }

    printf("----------------------------------\n");
    printf("Total Page Faults = %d\n", pageFaults);

    return 0;
}
