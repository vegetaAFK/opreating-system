#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX_LINE 1024

int main() {
    char filename[100], pattern[100];
    FILE *fp;
    char line[MAX_LINE];
    int lineNo = 0;
    int found = 0;

    printf("Enter file name: ");
    scanf("%s", filename);

    printf("Enter pattern to search: ");
    scanf("%s", pattern);

    fp = fopen(filename, "r");
    if (fp == NULL) {
        perror("Error opening file");
        return 1;
    }

    printf("\nMatching lines:\n");
    while (fgets(line, sizeof(line), fp) != NULL) {
        lineNo++;
        if (strstr(line, pattern) != NULL) {
            printf("%d: %s", lineNo, line);
            found = 1;
        }
    }

    if (!found)
        printf("Pattern not found in file.\n");

    fclose(fp);
    return 0;
}
