#include <stdio.h>

int main() {
    int files[50], indexBlock[50], n, start, length, i, j, k, c;

    for (i = 0; i < 50; i++)
        files[i] = 0; // All blocks are initially free

    do {
        printf("\nEnter the starting block and length of the file: ");
        scanf("%d %d", &start, &length);

        // Check if blocks are free
        k = 0;
        for (j = start; j < (start + length); j++) {
            if (files[j] == 0)
                k++;
        }

        if (k == length) {
            for (j = start; j < (start + length); j++) {
                files[j] = 1; // Mark blocks as allocated
            }
            printf("The file is allocated to blocks: ");
            for (j = start; j < (start + length); j++) {
                printf("%d ", j);
            }
            printf("\n");
        } else {
            printf("The file cannot be allocated (blocks already occupied)\n");
        }

        printf("\nDo you want to enter more files? (1 for Yes / 0 for No): ");
        scanf("%d", &c);
    } while (c == 1);

    printf("\nFinal Memory Block Status:\n");
    for (i = 0; i < 50; i++)
        printf("Block %d: %s\n", i, files[i] ? "Occupied" : "Free");

    return 0;
}
