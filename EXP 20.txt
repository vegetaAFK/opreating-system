#include <stdio.h>
#include <string.h>

struct File {
    char fname[20];
};

struct UserDirectory {
    char uname[20];
    int fcount;
    struct File files[20];
} userDir[20];

int main() {
    int userCount = 0, choice, i, j, k, flag;
    char uname[20], fname[20];

    while (1) {

        printf("\n--- Two Level Directory Structure ---\n");
        printf("1. Create User Directory\n");
        printf("2. Create File\n");
        printf("3. Delete File\n");
        printf("4. Search File\n");
        printf("5. Display Files of a User\n");
        printf("6. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {

        case 1:
            printf("Enter user name: ");
            scanf("%s", uname);

            // Check if user exists
            flag = 0;
            for (i = 0; i < userCount; i++) {
                if (strcmp(userDir[i].uname, uname) == 0) {
                    flag = 1;
                    break;
                }
            }
            if (flag)
                printf("User already exists!\n");
            else {
                strcpy(userDir[userCount].uname, uname);
                userDir[userCount].fcount = 0;
                userCount++;
                printf("User directory created.\n");
            }
            break;

        case 2:
            printf("Enter user name: ");
            scanf("%s", uname);

            // Find user
            flag = -1;
            for (i = 0; i < userCount; i++) {
                if (strcmp(userDir[i].uname, uname) == 0) {
                    flag = i;
                    break;
                }
            }
            if (flag == -1) {
                printf("User does not exist!\n");
                break;
            }

            printf("Enter file name to create: ");
            scanf("%s", fname);

            // Check for duplicate file
            for (j = 0; j < userDir[flag].fcount; j++) {
                if (strcmp(userDir[flag].files[j].fname, fname) == 0) {
                    printf("File already exists!\n");
                    break;
                }
            }

            if (j == userDir[flag].fcount) {
                strcpy(userDir[flag].files[userDir[flag].fcount].fname, fname);
                userDir[flag].fcount++;
                printf("File created successfully.\n");
            }
            break;

        case 3:
            printf("Enter user name: ");
            scanf("%s", uname);

            // Find user
            flag = -1;
            for (i = 0; i < userCount; i++) {
                if (strcmp(userDir[i].uname, uname) == 0) {
                    flag = i;
                    break;
                }
            }
            if (flag == -1) {
                printf("User not found!\n");
                break;
            }

            printf("Enter file name to delete: ");
            scanf("%s", fname);

            // Search file
            for (j = 0; j < userDir[flag].fcount; j++) {
                if (strcmp(userDir[flag].files[j].fname, fname) == 0) {
                    // Replace deleted file with last file
                    strcpy(userDir[flag].files[j].fname,
                           userDir[flag].files[userDir[flag].fcount - 1].fname);
                    userDir[flag].fcount--;
                    printf("File deleted.\n");
                    break;
                }
            }
            if (j == userDir[flag].fcount)
                printf("File not found!\n");
            break;

        case 4:
            printf("Enter user name: ");
            scanf("%s", uname);

            flag = -1;
            for (i = 0; i < userCount; i++) {
                if (strcmp(userDir[i].uname, uname) == 0) {
                    flag = i;
                    break;
                }
            }
            if (flag == -1) {
                printf("User not found!\n");
                break;
            }

            printf("Enter file name to search: ");
            scanf("%s", fname);

            for (j = 0; j < userDir[flag].fcount; j++) {
                if (strcmp(userDir[flag].files[j].fname, fname) == 0) {
                    printf("File found.\n");
                    break;
                }
            }
            if (j == userDir[flag].fcount)
                printf("File not found.\n");
            break;

        case 5:
            printf("Enter user name: ");
            scanf("%s", uname);

            flag = -1;
            for (i = 0; i < userCount; i++) {
                if (strcmp(userDir[i].uname, uname) == 0) {
                    flag = i;
                    break;
                }
            }
            if (flag == -1) {
                printf("User not found!\n");
                break;
            }

            printf("\nFiles of user %s:\n", uname);
            for (j = 0; j < userDir[flag].fcount; j++)
                printf("%s\n", userDir[flag].files[j].fname);
            break;

        case 6:
            return 0;

        default:
            printf("Invalid choice!\n");
        }
    }
}
