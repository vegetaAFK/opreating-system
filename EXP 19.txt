#include <stdio.h>
#include <string.h>

struct Directory {
    char fname[20];
} dir[20];

int main() {
    int n, i, choice, flag;
    char name[20];

    n = 0;  // initially no files

    while (1) {
        printf("\n--- Single Level Directory ---\n");
        printf("1. Create File\n");
        printf("2. Delete File\n");
        printf("3. Search File\n");
        printf("4. Display Files\n");
        printf("5. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {

        case 1:
            printf("Enter file name: ");
            scanf("%s", name);

            // Check for duplicate
            flag = 0;
            for (i = 0; i < n; i++) {
                if (strcmp(dir[i].fname, name) == 0) {
                    flag = 1;
                    break;
                }
            }

            if (flag)
                printf("Error: File already exists!\n");
            else {
                strcpy(dir[n].fname, name);
                n++;
                printf("File created successfully.\n");
            }
            break;

        case 2:
            printf("Enter file name to delete: ");
            scanf("%s", name);

            flag = 0;
            for (i = 0; i < n; i++) {
                if (strcmp(dir[i].fname, name) == 0) {
                    flag = 1;

                    // Shift the remaining files
                    strcpy(dir[i].fname, dir[n - 1].fname);
                    n--;

                    printf("File deleted successfully.\n");
                    break;
                }
            }

            if (!flag)
                printf("File not found!\n");
            break;

        case 3:
            printf("Enter file name to search: ");
            scanf("%s", name);

            flag = 0;
            for (i = 0; i < n; i++) {
                if (strcmp(dir[i].fname, name) == 0) {
                    flag = 1;
                    printf("File found.\n");
                    break;
                }
            }

            if (!flag)
                printf("File not found.\n");
            break;

        case 4:
            printf("\nFiles in directory:\n");
            for (i = 0; i < n; i++)
                printf("%s\n", dir[i].fname);
            break;

        case 5:
            return 0;

        default:
            printf("Invalid choice!\n");
        }
    }
}
