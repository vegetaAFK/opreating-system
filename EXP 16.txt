#include <stdio.h>

int main() {
    int blockCount, processCount;

    printf("Enter number of memory blocks: ");
    scanf("%d", &blockCount);

    int blockSize[blockCount];
    printf("Enter sizes of each block:\n");
    for (int i = 0; i < blockCount; i++)
        scanf("%d", &blockSize[i]);

    printf("Enter number of processes: ");
    scanf("%d", &processCount);

    int processSize[processCount];
    printf("Enter sizes of each process:\n");
    for (int i = 0; i < processCount; i++)
        scanf("%d", &processSize[i]);

    int allocation[processCount];
    for (int i = 0; i < processCount; i++)
        allocation[i] = -1;   // initially no allocation

    // First Fit Allocation
    for (int i = 0; i < processCount; i++) {
        for (int j = 0; j < blockCount; j++) {
            if (blockSize[j] >= processSize[i]) {
                allocation[i] = j;               // allocate block j
                blockSize[j] -= processSize[i];  // reduce block size
                break;                           // stop at first fit
            }
        }
    }

    // Output Results
    printf("\nProcess No.\tProcess Size\tBlock Allocated\n");
    for (int i = 0; i < processCount; i++) {
        printf("%d\t\t%d\t\t", i+1, processSize[i]);
        if (allocation[i] != -1)
            printf("%d\n", allocation[i] + 1);
        else
            printf("Not Allocated\n");
    }

    return 0;
}
